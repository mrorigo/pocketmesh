Less Boilerplate, More Logic

Define nodes, connect them, and run. Simple as that.

import { Flow, BaseNode } from "pocketmesh";

// Define a node
class HelloNode extends BaseNode {
  async prepare(_shared, params) { return params.name ?? "World"; }
  
  async execute(name) { 
    return `Hello, ${name}!`; 
  }
  
  async finalize(shared, _prep, result) {
    shared.greeting = result;
    return "default";
  }
}

// Run the flow
const flow = new Flow(new HelloNode());
await flow.runLifecycle({}, { name: "Mesh" });
import { generateAgentCard, a2aServerHandler } from "pocketmesh/a2a";

// 1. Define Agent Identity
const agentCard = generateAgentCard({
  name: "Greeter Agent",
  skills: [{ id: "greet", inputModes: ["text"] }]
});

// 2. Serve with Express
app.use("/a2a", a2aServerHandler({
  flows: { greet: new GreetFlow() },
  agentCard
}));
import { createA2AClient } from "pocketmesh/a2a";

// 1. Connect to any A2A agent
const client = await createA2AClient("https://agent.example.com");

// 2. Send a message (Task)
const response = await client.sendMessage({
  message: {
    role: "user",
    parts: [{ kind: "text", text: "Hello!" }]
  }
});
